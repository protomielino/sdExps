INCLUDE(../../../cmake/macros.cmake)

PROJECT(nfs2ac)

SET(_TARGET_NAME ${SD_TOOLS_EXECPREFIX}nfs2ac)

SET(NFS2AC_SOURCES mainnfs2ac.cpp)

ENDIF(UNIX)

SD_INSTALL_FILES(BIN TARGETS ${_TARGET_NAME})
